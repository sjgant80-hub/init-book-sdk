Install

Quick start

import { createReader } from '@ai-native-solutions/init-book-sdk';

const r = await createReader({ useIDB: true });
const chapter = await r.open(1);
console.log(chapter.title);      // "The Beginning"
console.log(await r.progress()); // { read: 1, total: 9, percent: 11.1, ... }

const svg = r.diagram('bloom-radial', { bloom: [4,7,7,6,5,4,3], label: 'me' });
